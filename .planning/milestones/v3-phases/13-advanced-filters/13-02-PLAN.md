```typescript
"use client"

import { Switch as SwitchPrimitive } from "@base-ui/react/switch"

import { cn } from "@/lib/utils"

function Switch({
  className,
  ...props
}: React.ComponentProps<typeof SwitchPrimitive.Root>) {
  return (
    <SwitchPrimitive.Root
      data-slot="switch"
      className={cn(
        // Track styles — rounded pill, transition on checked state
        "peer relative inline-flex h-5 w-9 shrink-0 cursor-pointer items-center rounded-full border-2 border-transparent transition-colors",
        "focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 focus-visible:ring-offset-background",
        "disabled:cursor-not-allowed disabled:opacity-50",
        // data-[checked] is Base UI's state attribute (not aria-checked)
        "bg-input data-[checked]:bg-primary",
        className
      )}
      {...props}
    >
      <SwitchPrimitive.Thumb
        data-slot="switch-thumb"
        className={cn(
          "pointer-events-none block h-4 w-4 rounded-full bg-background shadow-lg ring-0 transition-transform",
          // Translate from left to right when checked
          "translate-x-0 data-[checked]:translate-x-4"
        )}
      />
    </SwitchPrimitive.Root>
  )
}

export { Switch }
```

Key rules:
- Use `"use client"` (required — Switch is interactive)
- Import `React` is NOT needed with the modern JSX transform (Next.js 16); use `React.ComponentProps<...>` only if React is already imported. If `React` is not imported, type the props inline as `{ className?: string } & SwitchPrimitive.Root.Props` using the Base UI `.Props` type if it resolves — otherwise fall back to `React.ComponentProps` and add `import * as React from "react"`.
- Do NOT import from `@radix-ui/*` — only from `@base-ui/react/switch`
- Use `cn()` from `@/lib/utils` for all className merging
- Named export at bottom (not default export)
- The `data-[checked]` attribute is the correct Base UI 1.x state selector (not `data-[state=checked]` which is Radix syntax)

```typescript
"use client"

import * as React from "react"
import { Tooltip as TooltipPrimitive } from "@base-ui/react/tooltip"

import { cn } from "@/lib/utils"

function TooltipProvider({ ...props }: React.ComponentProps<typeof TooltipPrimitive.Provider>) {
  return <TooltipPrimitive.Provider data-slot="tooltip-provider" {...props} />
}

function Tooltip({ ...props }: React.ComponentProps<typeof TooltipPrimitive.Root>) {
  return <TooltipPrimitive.Root data-slot="tooltip" {...props} />
}

function TooltipTrigger({ ...props }: React.ComponentProps<typeof TooltipPrimitive.Trigger>) {
  return <TooltipPrimitive.Trigger data-slot="tooltip-trigger" {...props} />
}

function TooltipPopup({
  className,
  ...props
}: React.ComponentProps<typeof TooltipPrimitive.Popup>) {
  return (
    <TooltipPrimitive.Positioner>
      <TooltipPrimitive.Popup
        data-slot="tooltip-popup"
        className={cn(
          "z-50 overflow-hidden rounded-md bg-popover px-3 py-1.5 text-xs text-popover-foreground shadow-md",
          "animate-in fade-in-0 zoom-in-95",
          className
        )}
        {...props}
      />
    </TooltipPrimitive.Positioner>
  )
}

export {
  TooltipProvider,
  Tooltip,
  TooltipTrigger,
  TooltipPopup,
}
```

Key rules:
- `import * as React from "react"` is required here because `React.ComponentProps` is used throughout
- Do NOT import from `@radix-ui/*` — only from `@base-ui/react/tooltip`
- `TooltipPrimitive.Provider` provides delay and tooltip side-effects context; it is required (Pitfall 5 in RESEARCH.md)
- `TooltipPopup` wraps both `Positioner` and `Popup` together — this is the same convenience pattern as sheet.tsx wrapping `Portal` + `Overlay` + `Popup` in `SheetContent`
- Named exports at bottom (not default exports)
- If `TooltipPrimitive.Provider` does not exist as a sub-component on the Tooltip namespace (verify from node_modules), fall back to importing `{ TooltipProvider as TooltipProviderPrimitive } from "@base-ui/react/tooltip"` if exported at top level — check the actual exports before assuming

**Fallback if Provider sub-component prop type is unavailable:** Use `React.ComponentPropsWithoutRef<'div'>` or simply `{ children: React.ReactNode; delay?: number }` as an emergency inline type. The shape is less important than the file compiling cleanly.
